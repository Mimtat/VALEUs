document.addEventListener("DOMContentLoaded", () => { fetch("PrjAdd-2509.json") .then(r => r.json()) .then(data => { const g = new Graph(); // Color map per metaType const colorMap = { Country: "rgb(102, 255, 255)", Project: "rgb(255, 102, 102)", Organization: "rgb(102, 102, 255)", Addressee: "rgb(255, 255, 102)", "Action Domain": "rgb(255, 102, 255)" }; // Add nodes with mapped color data.nodes.forEach(n => { const m = n.attributes.metaType || ""; const col = colorMap[m] || "#999999"; g.addNode(n.key, { label: n.attributes.label, x: Number(n.attributes.x), y: Number(n.attributes.y), size: Number(n.attributes.size), color: col, metaType: m, renderType: "default" }); }); // Add edges data.edges.forEach((e, i) => { if (!g.hasEdge(e.source, e.target)) { g.addEdgeWithKey(e.key || "e" + i, e.source, e.target, { size: Number(e.attributes?.weight) || 1, color: e.attributes?.color || "#ccc", label: e.attributes?.relation || "" }); } }); const container = document.getElementById("container"); const renderer = new Sigma(g, container, { allowInvalidContainer: true }); // ForceAtlas2 continuous let running = true; const settings = { slowDown: 50 }; function loop() { if (running) { forceAtlas2.assign(g, { iterations: 1, settings }); } requestAnimationFrame(loop); } requestAnimationFrame(loop); // Pause/Resume const btn = document.getElementById("toggle"); btn.onclick = () => { running = !running; btn.textContent = running ? "⏸ Pause layout" : "▶ Resume layout"; }; // Filtering, but “Project” always active let activeTypes = new Set(["Country","Project","Addressee","Organization","Action Domain"]); document.querySelectorAll("#filters input").forEach(cb => { // disable Project checkbox if (cb.value === "Project") { cb.disabled = true; } cb.addEventListener("change", () => { if (cb.checked) activeTypes.add(cb.value); else activeTypes.delete(cb.value); renderer.refresh(); }); }); renderer.setSetting("nodeReducer", (node, data) => { const res = { ...data }; if (!activeTypes.has(data.metaType)) { res.hidden = true; } return res; }); renderer.setSetting("edgeReducer", (edge, data) => { const res = { ...data }; const src = g.source(edge); const tgt = g.target(edge); const srcType = g.getNodeAttribute(src, "metaType"); const tgtType = g.getNodeAttribute(tgt, "metaType"); if (!activeTypes.has(srcType) || !activeTypes.has(tgtType)) { res.hidden = true; } return res; }); }); });
